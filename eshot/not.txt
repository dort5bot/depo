function sadeceDurakSorgu(durak) {
    const form = document.createElement('form');
    form.method = 'POST';
    form.action = "https://www.eshot.gov.tr/tr/DuraktanGecenHatlar/" + durak;
    form.target = "_blank";

    const input1 = document.createElement('input');
    input1.type = 'hidden';
    input1.name = 'durakAdi';
    input1.value = durak;

    const input2 = document.createElement('input');
    input2.type = 'hidden';
    input2.name = 'durakId';
    input2.value = durak;

    form.appendChild(input1);
    form.appendChild(input2);

    document.body.appendChild(form);
    form.submit();
    document.body.removeChild(form);
}
